# Basic Highchart line chart
highchart() %>%
  hc_chart(type = "line") %>%
  hc_title(text = "Basic Line Chart") %>%
  hc_xAxis(categories = c("Jan", "Feb", "Mar")) %>%
  hc_add_series(name = "Apples", data = c(1, 4, 3)) %>%
  hc_add_series(name = "Bananas", data = c(5, 7, 3))

  hc_title(text = "Pie Chart Example") %>%

highchart() %>%
  hc_chart(type = "solidgauge") %>%
  hc_title(text = "Static Gauge Example") %>%
  hc_pane(
    center = list("50%", "85%"),
    size = "140%",
    startAngle = -90,
    endAngle = 90,
    background = list(
      backgroundColor = "#EEE",
      innerRadius = "60%",
      outerRadius = "100%",
      shape = "arc"
    )
  ) %>%
  hc_yAxis(
    min = 0, max = 5, title = list(text = "Score"),
    stops = list(
      list(0.2, "#DF5353"),
      list(0.5, "#DDDF0D"),
      list(0.8, "#55BF3B")
    ),
    lineWidth = 0, tickInterval = 1,
    labels = list(y = 16)
  ) %>%
  hc_add_series(
    name = "Score",
    data = list(3.7),
    tooltip = list(valueSuffix = " / 5")
  )
highchart() %>%
  hc_chart(type = "columnrange", inverted = TRUE) %>%
  hc_title(text = "Column Range Example") %>%
  hc_xAxis(categories = c("Jan", "Feb", "Mar")) %>%
  hc_yAxis(title = list(text = "Temperature (°C)")) %>%
  hc_add_series(
    name = "Temperatures",
    data = list(
      list(-9.7, 9.4),
      list(-8.7, 6.5),
      list(-3.5, 9.4)
    )
  )
